public void TrapInstantiate()
{
	if (catPlayer == null)
		catPlayer = GameObject.FindGameObjectWithTag("Player");

	Vector3 spawnLoc = new Vector3(catPlayer.transform.position.x, catPlayer.transform.position.y, catPlayer.transform.position.z);
	Quaternion spawnRotation = Quaternion.Euler(-90, 0, 0);

	if (PhotonNetwork.LocalPlayer.IsMasterClient)
	{
		PhotonNetwork.InstantiateRoomObject(trapPrefab.name, spawnLoc, spawnRotation);
	}

	CatTrappedScript.instance.trappingCat = true; 
	EscapeSpamScript.instance.startEscaping = true; // Enables on the cat player's buttom spam canvas
	trapNow = false;
}
